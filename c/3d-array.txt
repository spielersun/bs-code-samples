#include <stdio.h>
int main(void){
     int matrix[3][2][4] = {
          {
               { 14, 21, 12, 33 },
               { 56, 22, 15, 14 }
          },
          {
               { 23, 10, 99, 13 },
               { 15, 33, 21, 35 }
          },
          {
               { 78, 91, 11, 40 },
               { 22, 16, 53, 85 }
          }     
      };
      int toplam = 0;
      for(int x = 0 ; x < 3 ; ++x){
            for(int y = 0 ; y < 2 ; ++y){
                  for(int z = 0 ; z < 4 ; ++z){
                        toplam += matrix[x][y][z];
                  }
            }
      }
      printf("3 boyutlu listenin sayilari toplami %d.", toplam);
      return 0;
}
